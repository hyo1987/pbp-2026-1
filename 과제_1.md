### 최종 결과물 코드

- AI가 생성한 전체 코드 첨부

```c
#include <stdio.h>
#include <string.h>

#define MAX 100

typedef struct {
    char title[50];
    char date[20];
    char time[20];
    char desc[100];
} Schedule;

Schedule list[MAX];
int count = 0;

void addSchedule() {
    if (count >= MAX) {
        printf("Cannot add more schedules.\n");
        return;
    }

    printf("Title: ");
    scanf(" %[^\n]", list[count].title);

    printf("Date: ");
    scanf(" %[^\n]", list[count].date);

    printf("Time: ");
    scanf(" %[^\n]", list[count].time);

    printf("Description: ");
    scanf(" %[^\n]", list[count].desc);

    count++;
    printf("Schedule added successfully.\n");
}

void viewSchedule() {
    if (count == 0) {
        printf("No schedules saved.\n");
        return;
    }

    printf("\n===== Schedule List =====\n");

    for (int i = 0; i < count; i++) {
        printf("\n[%d]\n", i + 1);
        printf("Title: %s\n", list[i].title);
        printf("Date: %s\n", list[i].date);
        printf("Time: %s\n", list[i].time);
        printf("Description: %s\n", list[i].desc);
    }
}

void editSchedule() {
    int num;

    viewSchedule();

    if (count == 0) return;

    printf("\nEnter the schedule number to edit: ");
    scanf("%d", &num);

    if (num < 1 || num > count) {
        printf("Invalid number.\n");
        return;
    }

    num--;

    printf("New Title: ");
    scanf(" %[^\n]", list[num].title);

    printf("New Date: ");
    scanf(" %[^\n]", list[num].date);

    printf("New Time: ");
    scanf(" %[^\n]", list[num].time);

    printf("New Description: ");
    scanf(" %[^\n]", list[num].desc);

    printf("Schedule updated successfully.\n");
}

void deleteSchedule() {
    int num;

    viewSchedule();

    if (count == 0) return;

    printf("\nEnter the schedule number to delete: ");
    scanf("%d", &num);

    if (num < 1 || num > count) {
        printf("Invalid number.\n");
        return;
    }

    num--;

    for (int i = num; i < count - 1; i++) {
        list[i] = list[i + 1];
    }

    count--;

    printf("Schedule deleted successfully.\n");
}

int main() {
    int menu;

    while (1) {
        printf("\n===== Schedule Management Program =====\n");
        printf("1. Add Schedule\n");
        printf("2. View Schedules\n");
        printf("3. Edit Schedule\n");
        printf("4. Delete Schedule\n");
        printf("5. Exit\n");
        printf("Select menu: ");
        scanf("%d", &menu);

        switch (menu) {
            case 1:
                addSchedule();
                break;
            case 2:
                viewSchedule();
                break;
            case 3:
                editSchedule();
                break;
            case 4:
                deleteSchedule();
                break;
            case 5:
                printf("Program terminated.\n");
                return 0;
            default:
                printf("Invalid input.\n");
        }
    }

    return 0;
}

```
